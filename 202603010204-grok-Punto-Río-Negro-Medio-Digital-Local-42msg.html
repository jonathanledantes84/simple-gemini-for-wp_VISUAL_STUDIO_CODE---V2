https://puntorionegro.com.ar/ es el sitio web de PUNTO. RÍO NEGRO, un medio digital de noticias que se presenta como "El punto de encuentro con la información de Río Negro".

Se trata de un portal de noticias local/provincial enfocado principalmente en la provincia de Río Negro (Argentina, Patagonia), con cobertura de temas de actualidad, política, sociedad, economía, energía, turismo, cultura, tecnología, seguridad, policiales y más. Está organizado en secciones como:

  • Actualidad → Últimas noticias, Política, Sociedad
  • Economía y Recursos → Economía, Energía, Turismo
  • Sociedad y Cultura → Cultura, Tecnología
  • Opinión
  • Secciones regionales (ej. Alto Valle) y otras como Seguridad y Servicios, Policiales, etc.

El estilo parece ser de periodismo local estándar: mezcla de noticias serias (judiciales, política, economía) con temas cotidianos y regionales. Incluye opción de periodismo ciudadano ("Enviar noticia").

Importante aclaración: Este sitio no está relacionado con la iniciativa oficial del gobierno de Río Negro llamada "Punto Río Negro" (una marca de promoción de productos, turismo y productores rionegrinos impulsada por la Agencia de Desarrollo Económico RN, que aparece mucho en búsquedas oficiales y redes del gobierno provincial). Son dos proyectos distintos que comparten nombre similar.

Dado que estás en General Roca (zona Alto Valle de Río Negro), este medio probablemente cubre bastante información relevante para tu área, como noticias del Alto Valle, Valle Medio, judiciales locales, etc.

Si querés que mire alguna nota específica del sitio, que busque algo en particular (ej. noticias de hoy en Roca o Bariloche), o que compare con otros medios como Río Negro o LM Neuquén, decime! 😊

function simple_gemini_generate_text($prompt, $model = 'gemini-2.5-flash') {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key');

    $body = [
        'contents' => [['parts' => [['text' => $prompt]]]],
        'generationConfig' => ['temperature' => 0.7, 'maxOutputTokens' => 1200]
    ];

    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 90];
    $url = "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$key}";

    $response = wp_remote_post($url, $args);
    if (is_wp_error($response)) return $response;

    $code = wp_remote_retrieve_response_code($response);
    $body_resp = wp_remote_retrieve_body($response);

    if ($code < 200 || $code >= 300) return new WP_Error('api_error', "HTTP $code: " . substr($body_resp,0,300));

    $data = json_decode($body_resp, true);
    $text = $data['candidates'][0]['content']['parts'][0]['text'] ?? '';
    return $text ?: new WP_Error('no_text', 'No generó texto');
}

    $base64 = $data['candidates'][0]['content']['parts'][0]['inlineData']['data'] ?? '';

    $lines = explode("\n", trim($text));
    $title = strpos($lines[0], 'Título:') === 0 ? trim(substr($lines[0], 7)) : 'Noticia Local ' . date('d/m/Y');
    $content = implode("\n", array_slice($lines, strpos($lines[0], 'Título:') === 0 ? 1 : 0));

        'post_title' => sanitize_text_field($title),
        'post_content' => wp_kses_post($content),
        'post_status' => 'draft',
        'post_author' => 1,
        'post_category' => [get_cat_ID('Noticias') ?: 1],
        'featured_media' => is_wp_error($image_id) ? 0 : $image_id,
    ]);

            error_log('[Simple Gemini TG] Notificación enviada OK para post ' . $post_id);
        }
    }
}

                        ['type' => 'text', 'text' => esc_url($edit_link)],
                    ]
                ]
            ]
        ]
    ];

    $headers = [
        'Authorization' => 'Bearer ' . $token,
        'Content-Type'  => 'application/json',
    ];

    $max_retries = 2;
    $attempt = 0;
    $last_error = null;

    while ($attempt <= $max_retries) {
        $attempt++;

        $args = [
            'headers' => $headers,
            'body'    => wp_json_encode($body),
            'timeout' => 25,
        ];

        $response = wp_remote_post($url, $args);

        if (is_wp_error($response)) {
            $last_error = $response->get_error_message();
            error_log("[Simple Gemini WA] Intento $attempt/{$max_retries} - wp_remote_post falló: {$last_error}");
            
            if ($attempt <= $max_retries) {
                sleep(2); // backoff simple
                continue;
            }
            break;
        }

        $code = wp_remote_retrieve_response_code($response);
        $body_resp = wp_remote_retrieve_body($response);

        if ($code >= 200 && $code < 300) {
            error_log("[Simple Gemini WA] ✅ ÉXITO - Post ID {$post_id} enviado por WhatsApp");
            return true; // éxito total
        }

        // Manejo específico de errores Meta
        $decoded = json_decode($body_resp, true);
        $error_msg = $decoded['error']['message'] ?? $body_resp;
        $error_code = $decoded['error']['code'] ?? $code;

        error_log("[Simple Gemini WA] Intento $attempt - Error Meta {$error_code}: {$error_msg}");

        if (in_array($error_code, [4, 429, 500, 503])) { // rate limit o temporales
            if ($attempt <= $max_retries) {
                sleep(3 * $attempt); // backoff progresivo
                continue;
            }
        }

        $last_error = "Meta error {$error_code}: {$error_msg}";
        break; // error permanente → salimos
    }

    // Fallback automático a Telegram (si está activado)
    if (get_option('simple_gemini_tg_enable', 0)) {
        error_log('[Simple Gemini WA] → Intentando fallback Telegram');
        simple_gemini_send_telegram($title, $post_id, $edit_link, $image_id);
    }

    // Último fallback: Email
    simple_gemini_send_alert_email($post_id, $title, $edit_link);

    return new WP_Error('wa_failed', 'WhatsApp falló después de ' . $attempt . ' intentos: ' . $last_error);
}

function simple_gemini_generate_text($prompt, $model = 'gemini-2.5-flash') {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key Gemini');

    $body = ['contents' => [['parts' => [['text' => $prompt]]]]];
    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 90];

    $response = wp_remote_post("https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$key}", $args);

    if (is_wp_error($response)) return $response;

    $data = json_decode(wp_remote_retrieve_body($response), true);
    $text = $data['candidates'][0]['content']['parts'][0]['text'] ?? '';
    return $text ? $text : new WP_Error('no_text', 'No se generó texto');
}

// Generar texto (robusto)
function simple_gemini_generate_text($prompt, $model = 'gemini-2.5-flash') {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key Gemini');

    $body = ['contents' => [['parts' => [['text' => $prompt]]]]];
    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 90];

    $response = wp_remote_post("https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$key}", $args);

    if (is_wp_error($response)) {
        error_log('[Gemini Text] wp_remote_post error: ' . $response->get_error_message());
        return $response;
    }

    $code = wp_remote_retrieve_response_code($response);
    if ($code < 200 || $code >= 300) {
        error_log('[Gemini Text] HTTP ' . $code . ' - ' . wp_remote_retrieve_body($response));
        return new WP_Error('api_error', 'Gemini respondió con error HTTP ' . $code);
    }

    $data = json_decode(wp_remote_retrieve_body($response), true);
    $text = $data['candidates'][0]['content']['parts'][0]['text'] ?? '';
    return $text ?: new WP_Error('no_content', 'No se generó texto válido');
}

// Generar imagen (robusto)
function simple_gemini_generate_image($prompt, $model = 'gemini-2.5-flash-image') {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key');

    $body = ['contents' => [['parts' => [['text' => $prompt]]]], 'generationConfig' => ['responseModalities' => ['IMAGE']]];
    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 120];

    $response = wp_remote_post("https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$key}", $args);

    if (is_wp_error($response)) return $response;

    $data = json_decode(wp_remote_retrieve_body($response), true);
    $base64 = $data['candidates'][0]['content']['parts'][0]['inlineData']['data'] ?? '';

    if (empty($base64)) return new WP_Error('no_image', 'No se generó imagen');

    $upload = wp_upload_bits('gemini-' . time() . '.png', null, base64_decode($base64));
    if ($upload['error']) return new WP_Error('upload_error', $upload['error']);

    $attach_id = wp_insert_attachment([
        'post_mime_type' => 'image/png',
        'post_title'     => sanitize_file_name($prompt),
        'post_status'    => 'inherit'
    ], $upload['file']);

    require_once(ABSPATH . 'wp-admin/includes/image.php');
    wp_update_attachment_metadata($attach_id, wp_generate_attachment_metadata($attach_id, $upload['file']));

    return $attach_id;
}

// 1. WhatsApp Meta (principal)
function simple_gemini_send_whatsapp_meta($title, $post_id, $edit_link, $image_id = 0) {
    if (!get_option('simple_gemini_wa_enable', 0)) return new WP_Error('disabled', 'WhatsApp desactivado');

    $token    = get_option('simple_gemini_wa_token', '');
    $phone_id = get_option('simple_gemini_wa_phone_id', '');
    $to       = get_option('simple_gemini_wa_to', '');

    if (empty($token) || empty($phone_id) || empty($to)) {
        error_log('[WA Meta] ERROR: Faltan credenciales');
        return new WP_Error('config_missing', 'Faltan credenciales WhatsApp Meta');
    }

    $to = 'whatsapp:' . preg_replace('/\D/', '', $to);

    $url = "https://graph.facebook.com/v20.0/{$phone_id}/messages";

    $body = [
        'messaging_product' => 'whatsapp',
        'to'                => $to,
        'type'              => 'template',
        'template'          => [
            'name'       => 'nuevo_post_creado',
            'language'   => ['code' => 'es_AR'],
            'components' => [[
                'type'       => 'body',
                'parameters' => [
                    ['type' => 'text', 'text' => sanitize_text_field($title)],
                    ['type' => 'text', 'text' => esc_url_raw($edit_link)]
                ]
            ]]
        ]
    ];

    $args = [
        'headers' => [
            'Authorization' => 'Bearer ' . $token,
            'Content-Type'  => 'application/json'
        ],
        'body'    => wp_json_encode($body),
        'timeout' => 25
    ];

    $max_retries = 2;
    for ($attempt = 1; $attempt <= $max_retries; $attempt++) {
        $response = wp_remote_post($url, $args);
        if (!is_wp_error($response)) {
            $code = wp_remote_retrieve_response_code($response);
            if ($code >= 200 && $code < 300) {
                error_log("[WA Meta] ÉXITO - Post {$post_id}");
                return true;
            }
        }
        error_log("[WA Meta] Intento {$attempt} falló: " . (is_wp_error($response) ? $response->get_error_message() : $code));
        sleep(3);
    }

    // Fallback a Telegram si WhatsApp falla
    if (get_option('simple_gemini_tg_enable', 0)) {
        simple_gemini_send_telegram($title, $post_id, $edit_link, $image_id);
    }

    // Último fallback: Email
    simple_gemini_send_alert_email($post_id, $title, $edit_link);

    return new WP_Error('wa_failed', 'WhatsApp falló después de reintentos');
}

// 2. Telegram con sendPhoto (fallback)
function simple_gemini_send_telegram($title, $post_id, $edit_link, $image_id = 0) {
    if (!get_option('simple_gemini_tg_enable', 0)) return;

    $token   = get_option('simple_gemini_tg_token', '');
    $chat_id = get_option('simple_gemini_tg_chat_id', '');
    if (empty($token) || empty($chat_id)) return;

    $caption = "<b>Nuevo post IA!</b>\n<b>{$title}</b>\nFecha: " . date_i18n('d/m/Y H:i') . "\nLink: {$edit_link}";

    $api = "https://api.telegram.org/bot{$token}/";

    if ($image_id > 0) {
        $image_path = get_attached_file($image_id);
        if (file_exists($image_path)) {
            $boundary = wp_generate_uuid4();
            $body  = "--{$boundary}\r\n";
            $body .= 'Content-Disposition: form-data; name="chat_id"' . "\r\n\r\n{$chat_id}\r\n";
            $body .= "--{$boundary}\r\n";
            $body .= 'Content-Disposition: form-data; name="caption"' . "\r\n\r\n{$caption}\r\n";
            $body .= "--{$boundary}\r\n";
            $body .= 'Content-Disposition: form-data; name="photo"; filename="' . basename($image_path) . '"' . "\r\n";
            $body .= "Content-Type: image/png\r\n\r\n" . file_get_contents($image_path) . "\r\n";
            $body .= "--{$boundary}--\r\n";

            $response = wp_remote_post($api . 'sendPhoto', [
                'headers' => ['Content-Type' => "multipart/form-data; boundary={$boundary}"],
                'body'    => $body,
                'timeout' => 30
            ]);

            if (!is_wp_error($response) && wp_remote_retrieve_response_code($response) === 200) return;
        }
    }

    // Fallback texto si imagen falla
    wp_remote_post($api . 'sendMessage', [
        'body' => [
            'chat_id'    => $chat_id,
            'text'       => $caption,
            'parse_mode' => 'HTML'
        ]
    ]);
}

        : "Genera nota sobre actualidad en General Roca / Alto Valle (turismo, economía, clima). Título al inicio.";

    $text = simple_gemini_generate_text($prompt);
    if (is_wp_error($text)) {
        error_log('[Auto Post] Falló generación texto: ' . $text->get_error_message());
        return;
    }

    $lines = explode("\n", trim($text));
    $title = (strpos($lines[0] ?? '', 'Título:') === 0) ? trim(substr($lines[0], 7)) : 'Actualidad IA - ' . date_i18n('d/m/Y');
    $content = implode("\n", array_slice($lines, strpos($lines[0] ?? '', 'Título:') === 0 ? 1 : 0));

    $image_id = simple_gemini_generate_image("Foto destacada realista: " . wp_trim_words($title . ' ' . $content, 15) . " General Roca Río Negro");

    $post_id = wp_insert_post([
        'post_title'     => sanitize_text_field($title),
        'post_content'   => wp_kses_post($content),
        'post_status'    => 'draft',
        'post_author'    => 1,
        'featured_media' => is_wp_error($image_id) ? 0 : $image_id,
    ], true);

    if (is_wp_error($post_id)) {
        error_log('[Auto Post] Falló wp_insert_post: ' . $post_id->get_error_message());
        return;
    }

    // Categorías automáticas
    $cats = [get_cat_ID('Noticias Automáticas') ?: 1];
    $lower = strtolower($content);
    if (strpos($lower, 'turismo') !== false || strpos($lower, 'visitar') !== false) $cats[] = get_cat_ID('Turismo') ?: 0;
    if (strpos($lower, 'policial') !== false || strpos($lower, 'delito') !== false) $cats[] = get_cat_ID('Policiales') ?: 0;
    if (strpos($lower, 'econom') !== false || strpos($lower, 'precio') !== false) $cats[] = get_cat_ID('Economía') ?: 0;
    wp_set_post_categories($post_id, array_filter($cats));

    $edit_link = get_edit_post_link($post_id, '');
    simple_gemini_send_whatsapp_meta($title, $post_id, $edit_link, $image_id);
}

/**

 * Robusta: maneja errores por feed, limita ítems, logs detallados

function simple_gemini_get_x_tweets() {
    $urls = array_filter(array_map('trim', explode(',', get_option('simple_gemini_x_rss_urls', ''))));
    if (empty($urls)) return '';

    $content = '';
    $errors = [];

    foreach ($urls as $url) {
        $rss = fetch_feed(esc_url_raw($url));
        if (is_wp_error($rss)) {
            $err = 'fetch_feed falló para X RSS: ' . $rss->get_error_message() . ' (' . $url . ')';
            error_log('[Simple Gemini X] ' . $err);
            $errors[] = $err;
            continue;
        }

        if ($rss->error()) {
            $err = 'SimplePie error en X RSS: ' . $rss->error() . ' (' . $url . ')';
            error_log('[Simple Gemini X] ' . $err);
            $errors[] = $err;
            continue;
        }

        $items = $rss->get_items(0, 4); // Máx 4 tweets por feed para no abusar
        if (empty($items)) continue;

        foreach ($items as $item) {
            $title   = esc_html($item->get_title() ?: 'Tweet sin título');
            $link    = esc_url($item->get_permalink());
            $desc    = wp_strip_all_tags($item->get_content() ?: $item->get_description());
            $date    = $item->get_date('d/m/Y H:i') ?: 'Fecha desconocida';

            $content .= "Tweet: {$title} ({$date})\nEnlace: {$link}\nContenido: " . wp_trim_words($desc, 50, '...') . "\n\n";
        }
    }

    if (empty($content) && !empty($errors)) {
        error_log('[Simple Gemini X] Todos los feeds de X fallaron: ' . implode('; ', $errors));
    }

    return trim($content);
}
