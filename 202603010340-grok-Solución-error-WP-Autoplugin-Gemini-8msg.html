// ====================== FUNCIÓN PRINCIPAL ======================
function simple_gemini_process_feeds() {
    $options = get_option('simple_gemini_options');
    if (empty($options['api_key'])) return 'Falta API Key Gemini';

    $feeds = [
        'noticias' => array_filter(explode(',', $options['rss_noticias'] ?? '')),
        'x'        => array_filter(explode(',', $options['rss_x'] ?? '')),
        'ig'       => array_filter(explode(',', $options['rss_ig'] ?? ''))
    ];

    $items = [];
    foreach ($feeds as $type => $urls) {
        foreach ($urls as $url) {
            $rss = simplexml_load_file(trim($url));
            if ($rss) {
                foreach ($rss->channel->item as $item) {
                    $items[] = [
                        'title'       => (string)$item->title,
                        'description' => (string)$item->description,
                        'link'        => (string)$item->link,
                        'image'       => (string)($item->enclosure['url'] ?? ''),
                        'type'        => $type
                    ];
                }
            }
        }
    }

    if (empty($items)) return 'No se encontraron items en los RSS';

    // Tomamos el primero más reciente (podés mejorar con fecha)
    $item = $items[0];

    // Prompt personalizado
    $prompt = str_replace(
        ['{titulo}', '{descripcion}', '{fuente}'],
        [$item['title'], wp_strip_all_tags($item['description']), $item['link']],
        $options['custom_prompt'] ?? 'Escribí un artículo...'
    );

    // Llamada a Gemini (con Vision si hay imagen IG)
    $gemini_response = simple_gemini_call($prompt, $item['image'], $options['api_key']);

    if (is_wp_error($gemini_response)) return $gemini_response->get_error_message();

    // Crear post
    $post_id = wp_insert_post([
        'post_title'   => $gemini_response['title'],
        'post_content' => $gemini_response['content'],
        'post_status'  => 'draft',
        'post_category'=> [get_cat_ID('Sin categoría')], // podés agregar lógica de categorías
    ]);

    if ($post_id && !empty($gemini_response['featured_image_url'])) {
        // Descargar y setear imagen destacada
        $attach_id = simple_gemini_set_featured_image($gemini_response['featured_image_url'], $post_id);
    }

    // Notificación
    simple_gemini_send_notification($post_id, $gemini_response['title'], $item['link']);

    return "Post creado correctamente (ID: $post_id)";
}

// ====================== GEMINI CALL (texto + Vision) ======================
function simple_gemini_call($prompt, $image_url = '', $api_key) {
    $url = "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=" . $api_key;

    $parts = [['text' => $prompt]];

    if ($image_url && filter_var($image_url, FILTER_VALIDATE_URL)) {
        $image_data = @file_get_contents($image_url);
        if ($image_data) {
            $base64 = base64_encode($image_data);
            $mime = mime_content_type($image_url) ?: 'image/jpeg';
            $parts[] = [
                'inline_data' => [
                    'mime_type' => $mime,
                    'data'      => $base64
                ]
            ];
        }
    }

    $body = json_encode([
        'contents' => [['parts' => $parts]]
    ]);

    $response = wp_remote_post($url, [
        'headers' => ['Content-Type' => 'application/json'],
        'body'    => $body,
        'timeout' => 45
    ]);

    if (is_wp_error($response)) return $response;

    $data = json_decode(wp_remote_retrieve_body($response), true);

    $text = $data['candidates'][0]['content']['parts'][0]['text'] ?? '';

    // Parsear título y contenido (Gemini suele responder con **Título:** ...)
    preg_match('/^Título[:\s]*(.+?)(?:\n|$)/mi', $text, $title);
    $title = $title[1] ?? 'Noticia IA - ' . date('d/m/Y');

    return [
        'title' => trim($title),
        'content' => $text,
        'featured_image_url' => '' // podés extender para que Gemini genere imagen con Imagen 3
    ];
}

// ====================== IMAGEN DESTACADA ======================
function simple_gemini_set_featured_image($url, $post_id) {
    $tmp = download_url($url);
    if (is_wp_error($tmp)) return false;

    $file_array = ['name' => basename($url), 'tmp_name' => $tmp];
    $id = media_handle_sideload($file_array, $post_id);

    if (!is_wp_error($id)) {
        set_post_thumbnail($post_id, $id);
    }
    @unlink($tmp);
    return $id;
}

// ====================== NOTIFICACIONES ======================
function simple_gemini_send_notification($post_id, $title, $link) {
    $options = get_option('simple_gemini_options');
    $message = "📰 Nuevo post en Punto Río Negro\n\n$title\n\n$link";

    // WhatsApp primero (Meta Cloud API)
    if (!empty($options['whatsapp_phone_id']) && !empty($options['whatsapp_token'])) {
        $wa_url = "https://graph.facebook.com/v20.0/{$options['whatsapp_phone_id']}/messages";
        wp_remote_post($wa_url, [
            'headers' => [
                'Authorization' => 'Bearer ' . $options['whatsapp_token'],
                'Content-Type'  => 'application/json'
            ],
            'body' => json_encode([
                'messaging_product' => 'whatsapp',
                'to' => $options['whatsapp_to'] ?? '', // agregá campo
                'type' => 'text',
                'text' => ['body' => $message]
            ])
        ]);
    }

    // Telegram fallback
    if (!empty($options['telegram_token']) && !empty($options['telegram_chat_id'])) {
        wp_remote_post("https://api.telegram.org/bot{$options['telegram_token']}/sendMessage", [
            'body' => [
                'chat_id' => $options['telegram_chat_id'],
                'text'    => $message,
                'parse_mode' => 'HTML'
            ]
        ]);
    }

    // Email último
    wp_mail($options['email'] ?? get_option('admin_email'), 'Nuevo post automático', $message);
}

function simple_gemini_call($prompt, $image_url = '', $api_key) {
    $options = get_option('simple_gemini_options');
    $model = $options['gemini_model'] ?? 'gemini-1.5-flash';

    $url = "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key=" . $api_key;

    $parts = [['text' => $prompt]];

    // Si hay imagen de IG/RSS → análisis Vision
    if ($image_url && filter_var($image_url, FILTER_VALIDATE_URL)) {
        $image_data = @file_get_contents($image_url);
        if ($image_data) {
            $base64 = base64_encode($image_data);
            $mime = @mime_content_type($image_url) ?: 'image/jpeg';
            $parts[] = [
                'inline_data' => [
                    'mime_type' => $mime,
                    'data'      => $base64
                ]
            ];
        }
    }

    // Prompt para generar imagen destacada si no hay
    $image_prompt = "Genera una imagen destacada profesional, fotorealista, estilo noticia local de Río Negro/Alto Valle, relacionada con: " . substr($prompt, 0, 200) . ". Formato horizontal, alta resolución, colores vibrantes patagónicos.";

    $body = [
        'contents' => [['parts' => $parts]],
        'generationConfig' => [
            'temperature' => 0.7,
            'topP' => 0.95,
            'maxOutputTokens' => 2048
        ]
    ];

    // Si queremos imagen → cambiamos el prompt y pedimos inline data (Imagen 3 mode)
    if (empty($image_url)) {  // Solo genera si no vino imagen del feed
        $body['contents'][0]['parts'][0]['text'] .= "\n\nAl final, genera una imagen destacada con Imagen 3 basada en el artículo.";
        // Nota: En 2026, para forzar imagen response, algunos usan tools o config específica
        // Si no funciona, fallback a describir y usar DALL-E externo, pero probemos
    }

    $response = wp_remote_post($url, [
        'headers' => ['Content-Type' => 'application/json'],
        'body'    => json_encode($body),
        'timeout' => 60
    ]);

    if (is_wp_error($response)) return $response;

    $data = json_decode(wp_remote_retrieve_body($response), true);

    if (empty($data['candidates'][0]['content']['parts'])) {
        return new WP_Error('gemini_error', 'No response from Gemini');
    }

    $text = '';
    $generated_image_base64 = null;
    $mime_type = null;

    foreach ($data['candidates'][0]['content']['parts'] as $part) {
        if (isset($part['text'])) {
            $text .= $part['text'];
        }
        if (isset($part['inline_data'])) {
            $generated_image_base64 = $part['inline_data']['data'];
            $mime_type = $part['inline_data']['mime_type'];
        }
    }

    // Parse título y contenido
    preg_match('/^Título[:\s]*(.+?)(?:\n|$)/mi', $text, $title_match);
    $title = $title_match[1] ?? 'Noticia IA - ' . date('d/m/Y');

    $featured_url = '';

    if ($generated_image_base64) {
        // Guardar imagen generada como attachment
        $upload_dir = wp_upload_dir();
        $file_name = 'gemini-featured-' . time() . '.png';
        $file_path = $upload_dir['path'] . '/' . $file_name;
        file_put_contents($file_path, base64_decode($generated_image_base64));

        $attachment = [
            'guid'           => $upload_dir['url'] . '/' . $file_name,
            'post_mime_type' => $mime_type ?: 'image/png',
            'post_title'     => $title,
            'post_content'   => '',
            'post_status'    => 'inherit'
        ];

        $attach_id = wp_insert_attachment($attachment, $file_path);
        if ($attach_id) {
            require_once(ABSPATH . 'wp-admin/includes/image.php');
            $attach_data = wp_generate_attachment_metadata($attach_id, $file_path);
            wp_update_attachment_metadata($attach_id, $attach_data);
            $featured_url = wp_get_attachment_url($attach_id);
        }
    }

    return [
        'title'              => trim($title),
        'content'            => $text,
        'featured_image_url' => $featured_url
    ];
}
