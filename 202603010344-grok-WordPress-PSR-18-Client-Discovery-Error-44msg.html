function simple_gemini_get_api_key() {
    if ( defined('SIMPLE_GEMINI_API_KEY') && SIMPLE_GEMINI_API_KEY ) {
        return SIMPLE_GEMINI_API_KEY;

// --- Función principal: genera texto con Gemini ---
function simple_gemini_generate_text( $prompt, $model = 'gemini-2.5-flash', $options = [] ) {
    $api_key = simple_gemini_get_api_key();
    if ( empty($api_key) ) {
        return new WP_Error('no_api_key', 'No API key configurada para Simple Gemini.');
    }

    // Formato de request para Gemini (contents como array de mensajes)
    $body = [
        'contents' => [
            [
                'parts' => [
                    ['text' => $prompt]
                ]
            ]
        ],
        'generationConfig' => array_merge([
            'temperature'     => 0.7,
            'topP'            => 0.95,
            'topK'            => 40,
            'maxOutputTokens' => 1200,
        ], $options['generationConfig'] ?? []),
    ];

    // Si querés forzar salida JSON (útil para datos estructurados)
    if ( ! empty( $options['response_mime_type'] ) ) {
        $body['generationConfig']['response_mime_type'] = $options['response_mime_type'];
        if ( ! empty( $options['response_schema'] ) ) {
            $body['generationConfig']['response_schema'] = $options['response_schema'];
        }
    }

    $args = [
        'headers' => [
            'Content-Type' => 'application/json',
        ],
        'body'    => wp_json_encode($body),
        'timeout' => 60,
    ];

    $url = "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$api_key}";

    $response = wp_remote_post( $url, $args );

    if ( is_wp_error($response) ) {
        return $response;
    }

    $code = wp_remote_retrieve_response_code($response);
    $body_resp = wp_remote_retrieve_body($response);

    if ( $code < 200 || $code >= 300 ) {
        return new WP_Error('api_error', "Código $code: " . substr($body_resp, 0, 500));
    }

    $data = json_decode($body_resp, true);
    if ( json_last_error() !== JSON_ERROR_NONE ) {
        return new WP_Error('json_error', 'Respuesta JSON inválida: ' . json_last_error_msg());
    }

    // Extraer el texto generado (Gemini devuelve candidates[0].content.parts[0].text)
    $text = '';
    if ( ! empty( $data['candidates'][0]['content']['parts'][0]['text'] ) ) {
        $text = $data['candidates'][0]['content']['parts'][0]['text'];
    }

    // Si es JSON mode, devolver el raw parsed si es válido
    if ( ! empty( $options['response_mime_type'] ) && $options['response_mime_type'] === 'application/json' ) {
        $json = json_decode($text, true);
        return $json ?: $text;
    }

    return $text !== '' ? $text : $data;
}

function simple_gemini_create_post( $prompt, $model = 'gemini-2.5-flash', $post_title = '' ) {
    $content = simple_gemini_generate_text( $prompt, $model );
    if ( is_wp_error($content) ) return $content;

    if ( empty($post_title) ) {
        $post_title = wp_trim_words( strip_tags($prompt), 8, '...' );
    }

    $post_id = wp_insert_post([
        'post_title'   => $post_title,
        'post_content' => wp_kses_post( $content ),
        'post_status'  => 'draft',
        'post_author'  => get_current_user_id(),
        'post_type'    => 'post',
    ]);

    return $post_id;
}

function simple_gemini_generate_image( $prompt, $model = 'gemini-3.1-flash-image-preview', $options = [] ) {
    $api_key = simple_gemini_get_api_key();
    if ( empty($api_key) ) return new WP_Error('no_api_key', 'No API key.');

    $body = [
        'contents' => [
            [
                'parts' => [
                    ['text' => $prompt]
                ]
            ]
        ],
        'generationConfig' => array_merge([
            'responseModalities' => ['image'],  // o ['text', 'image'] para combo
            'temperature' => 0.8,
        ], $options),
    ];

    $args = [
        'headers' => ['Content-Type' => 'application/json'],
        'body'    => wp_json_encode($body),
        'timeout' => 90,
    ];

    $url = "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$api_key}";

    $response = wp_remote_post( $url, $args );

    if ( is_wp_error($response) ) return $response;

    $code = wp_remote_retrieve_response_code($response);
    $body_resp = wp_remote_retrieve_body($response);

    if ( $code < 200 || $code >= 300 ) return new WP_Error('api_error', "Código $code: " . substr($body_resp, 0, 300));

    $data = json_decode($body_resp, true);
    $image_base64 = '';
    if ( ! empty( $data['candidates'][0]['content']['parts'][0]['inlineData']['data'] ) ) {
        $image_base64 = $data['candidates'][0]['content']['parts'][0]['inlineData']['data'];
    }

    if ( empty($image_base64) ) return new WP_Error('no_image', 'No imagen generada.');

    // Guardar como attachment en WP
    $upload = wp_upload_bits( 'gemini-image-' . time() . '.png', null, base64_decode($image_base64) );
    if ( $upload['error'] ) return new WP_Error('upload_error', $upload['error']);

    $attachment = [
        'post_mime_type' => 'image/png',
        'post_title'     => sanitize_file_name( $prompt ),
        'post_content'   => '',
        'post_status'    => 'inherit'
    ];
    $attach_id = wp_insert_attachment( $attachment, $upload['file'] );
    require_once( ABSPATH . 'wp-admin/includes/image.php' );
    $attach_data = wp_generate_attachment_metadata( $attach_id, $upload['file'] );
    wp_update_attachment_metadata( $attach_id, $attach_data );

    return $attach_id;  // Devuelve ID del attachment
}

// --- Obtener contenido reciente de RSS ---
function simple_gemini_get_rss_content() {
    $rss_urls = get_option('simple_gemini_rss_urls', '');
    if ( empty( $rss_urls ) ) {
        return ''; // No RSS → fallback a prompt fijo
    }

    $urls = array_map('trim', explode(',', $rss_urls)); // o explode("\n") si usás líneas
    $content = '';

    foreach ( $urls as $url ) {
        if ( empty($url) ) continue;

        $rss = fetch_feed( esc_url_raw( $url ) );
        if ( is_wp_error( $rss ) ) {
            error_log( 'Simple Gemini RSS Error: ' . $rss->get_error_message() . ' en ' . $url );
            continue;
        }

        $max_items = $rss->get_item_quantity(3); // Limita a 3 para no abusar
        $items = $rss->get_items(0, $max_items);

        if ( empty( $items ) ) continue;

        // Toma el más reciente
        $item = $items[0];
        $title = $item->get_title();
        $link = $item->get_permalink();
        $desc = $item->get_description(); // o get_content() si full text
        $date = $item->get_date('d/m/Y H:i');

        $content .= "Noticia reciente: $title ($date)\nEnlace: $link\nResumen/Contenido: $desc\n\n";
    }

    return trim( $content );
}

function simple_gemini_generate_text( $prompt, $model = 'gemini-2.5-flash', $options = [] ) {
    $api_key = simple_gemini_get_api_key();
    if ( empty( $api_key ) ) {
        error_log( '[Simple Gemini] Error: No API key configurada.' );
        return new WP_Error( 'no_api_key', 'No API key configurada para Simple Gemini. Verifica en Ajustes.' );
    }

    $body = [
        'contents' => [
            [
                'parts' => [ ['text' => $prompt] ]
            ]
        ],
        'generationConfig' => array_merge( [
            'temperature'     => 0.7,
            'topP'            => 0.95,
            'topK'            => 40,
            'maxOutputTokens' => 1200,
        ], $options['generationConfig'] ?? [] ),
    ];

    if ( ! empty( $options['response_mime_type'] ) ) {
        $body['generationConfig']['response_mime_type'] = $options['response_mime_type'];
        if ( ! empty( $options['response_schema'] ) ) {
            $body['generationConfig']['response_schema'] = $options['response_schema'];
        }
    }

    $args = [
        'headers' => [ 'Content-Type' => 'application/json' ],
        'body'    => wp_json_encode( $body ),
        'timeout' => 90,          // Aumentado para respuestas largas
        'sslverify' => true,      // Seguridad por defecto
    ];

    $url = "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$api_key}";

    $response = wp_remote_post( $url, $args );

    if ( is_wp_error( $response ) ) {
        $error_msg = 'wp_remote_post falló: ' . $response->get_error_message();
        error_log( '[Simple Gemini API] ' . $error_msg );
        return new WP_Error( 'http_request_failed', $error_msg );
    }

    $code = wp_remote_retrieve_response_code( $response );
    $body_resp = wp_remote_retrieve_body( $response );

    if ( $code < 200 || $code >= 300 ) {
        $msg = "Gemini API devolvió error HTTP $code: " . substr( $body_resp, 0, 300 );
        error_log( '[Simple Gemini API] ' . $msg );
        return new WP_Error( 'api_http_error', $msg, [ 'code' => $code, 'response' => $body_resp ] );
    }

    $data = json_decode( $body_resp, true );
    if ( json_last_error() !== JSON_ERROR_NONE ) {
        $msg = 'Respuesta JSON inválida: ' . json_last_error_msg();
        error_log( '[Simple Gemini API] ' . $msg );
        return new WP_Error( 'json_decode_failed', $msg );
    }

    // Manejo de errores comunes de Gemini (ej: rate limit, safety block)
    if ( isset( $data['error'] ) ) {
        $msg = 'Gemini error: ' . ( $data['error']['message'] ?? 'Desconocido' );
        error_log( '[Simple Gemini API] ' . $msg );
        return new WP_Error( 'gemini_api_error', $msg, $data['error'] );
    }

    $text = $data['candidates'][0]['content']['parts'][0]['text'] ?? '';
    if ( empty( $text ) ) {
        error_log( '[Simple Gemini API] No se generó texto válido en la respuesta.' );
        return new WP_Error( 'no_content_generated', 'No se generó texto válido.' );
    }

    return $text;
}

function simple_gemini_get_rss_content() {
    $rss_urls = get_option( 'simple_gemini_rss_urls', '' );
    if ( empty( $rss_urls ) ) {
        return ''; // Fallback silencioso
    }

    $urls = array_filter( array_map( 'trim', explode( ',', $rss_urls ) ) );
    if ( empty( $urls ) ) return '';

    $content = '';
    $errors  = [];

    foreach ( $urls as $url ) {
        $rss = fetch_feed( esc_url_raw( $url ) );

        if ( is_wp_error( $rss ) ) {
            $err = 'fetch_feed falló para ' . $url . ': ' . $rss->get_error_message();
            error_log( '[Simple Gemini RSS] ' . $err );
            $errors[] = $err;
            continue;
        }

        if ( $rss->error() ) {
            $err = 'SimplePie error en ' . $url . ': ' . $rss->error();
            error_log( '[Simple Gemini RSS] ' . $err );
            $errors[] = $err;
            continue;
        }

        $items = $rss->get_items( 0, 3 ); // Limita fuertemente
        if ( empty( $items ) ) continue;

        foreach ( $items as $item ) {
            $title = esc_html( $item->get_title() ?: 'Sin título' );
            $link  = esc_url( $item->get_permalink() );
            $desc  = $item->get_content() ?: $item->get_description(); // Prefiere full content
            $date  = $item->get_date( 'd/m/Y H:i' ) ?: 'Fecha desconocida';

            $content .= "Noticia: $title ($date)\nEnlace: $link\nContenido: " . wp_strip_all_tags( $desc ) . "\n\n";
        }
    }

    if ( empty( $content ) && ! empty( $errors ) ) {
        error_log( '[Simple Gemini RSS] Todos los feeds fallaron: ' . implode( '; ', $errors ) );
    }

    return trim( $content );
}

function simple_gemini_generate_text($prompt, $model = 'gemini-2.5-flash') {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key Gemini');
    $body = ['contents' => [['parts' => [['text' => $prompt]]]]];
    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 90];
    $response = wp_remote_post("https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent?key={$key}", $args);
    if (is_wp_error($response)) return $response;
    $data = json_decode(wp_remote_retrieve_body($response), true);
    return $data['candidates'][0]['content']['parts'][0]['text'] ?? new WP_Error('no_text', 'No se generó texto');
}

function simple_gemini_generate_image($prompt) {
    $key = simple_gemini_get_api_key();
    if (empty($key)) return new WP_Error('no_key', 'Falta API Key');
    $body = ['contents' => [['parts' => [['text' => $prompt]]]], 'generationConfig' => ['responseModalities' => ['IMAGE']]];
    $args = ['headers' => ['Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 120];
    $response = wp_remote_post("https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image:generateContent?key={$key}", $args);
    $data = json_decode(wp_remote_retrieve_body($response), true);
    $base64 = $data['candidates'][0]['content']['parts'][0]['inlineData']['data'] ?? '';
    if (empty($base64)) return new WP_Error('no_image', 'No se generó imagen');
    $upload = wp_upload_bits('gemini-' . time() . '.png', null, base64_decode($base64));
    if ($upload['error']) return new WP_Error('upload_error', $upload['error']);
    $attach_id = wp_insert_attachment(['post_mime_type' => 'image/png', 'post_title' => sanitize_file_name($prompt), 'post_status' => 'inherit'], $upload['file']);
    require_once(ABSPATH . 'wp-admin/includes/image.php');
    wp_update_attachment_metadata($attach_id, wp_generate_attachment_metadata($attach_id, $upload['file']));
    return $attach_id;
}

function simple_gemini_get_rss_content() {
    $urls = array_filter(array_map('trim', explode(',', get_option('simple_gemini_rss_urls', ''))));
    $content = '';
    foreach ($urls as $url) {
        $rss = fetch_feed(esc_url_raw($url));
        if (is_wp_error($rss)) continue;
        $items = $rss->get_items(0, 2);
        foreach ($items as $item) {
            $content .= "Noticia: " . esc_html($item->get_title()) . " - " . wp_strip_all_tags($item->get_content() ?: $item->get_description()) . "\n\n";
        }
    }
    return trim($content);
}

function simple_gemini_send_whatsapp_meta($title, $post_id, $edit_link, $image_id = 0) {
    if (!get_option('simple_gemini_wa_enable', 0)) return;
    $token = get_option('simple_gemini_wa_token','');
    $phone_id = get_option('simple_gemini_wa_phone_id','');
    $to = get_option('simple_gemini_wa_to','');
    if (empty($token) || empty($phone_id) || empty($to)) return;

    $to = 'whatsapp:' . preg_replace('/\D/', '', $to);
    $url = "https://graph.facebook.com/v20.0/{$phone_id}/messages";
    $body = [
        'messaging_product' => 'whatsapp',
        'to' => $to,
        'type' => 'template',
        'template' => [
            'name' => 'nuevo_post_creado',
            'language' => ['code' => 'es_AR'],
            'components' => [[
                'type' => 'body',
                'parameters' => [
                    ['type' => 'text', 'text' => sanitize_text_field($title)],
                    ['type' => 'text', 'text' => esc_url_raw($edit_link)]
                ]
            ]]
        ]
    ];

    $args = ['headers' => ['Authorization' => 'Bearer ' . $token, 'Content-Type' => 'application/json'], 'body' => wp_json_encode($body), 'timeout' => 25];

    for ($i = 1; $i <= 2; $i++) {
        $response = wp_remote_post($url, $args);
        if (!is_wp_error($response) && wp_remote_retrieve_response_code($response) >= 200 && wp_remote_retrieve_response_code($response) < 300) {
            error_log("[WA Meta] OK - Post {$post_id}");
            return;
        }
        sleep(2);
    }

    // Fallback Telegram
    if (get_option('simple_gemini_tg_enable', 0)) simple_gemini_send_telegram($title, $post_id, $edit_link, $image_id);
    // Fallback Email
    simple_gemini_send_alert_email($post_id, $title, $edit_link);
}

function simple_gemini_send_telegram($title, $post_id, $edit_link, $image_id = 0) {
    if (!get_option('simple_gemini_tg_enable', 0)) return;
    $token = get_option('simple_gemini_tg_token','');
    $chat_id = get_option('simple_gemini_tg_chat_id','');
    if (empty($token) || empty($chat_id)) return;

    $caption = "<b>Nuevo post IA!</b>\n<b>" . esc_html($title) . "</b>\nFecha: " . date_i18n('d/m/Y H:i') . "\nLink: " . esc_url($edit_link);

    if ($image_id > 0) {
        $image_path = get_attached_file($image_id);
        if (file_exists($image_path)) {
            $boundary = wp_generate_uuid4();
            $body = "--{$boundary}\r\nContent-Disposition: form-data; name=\"chat_id\"\r\n\r\n{$chat_id}\r\n";
            $body .= "--{$boundary}\r\nContent-Disposition: form-data; name=\"caption\"\r\n\r\n{$caption}\r\n";
            $body .= "--{$boundary}\r\nContent-Disposition: form-data; name=\"photo\"; filename=\"" . basename($image_path) . "\"\r\nContent-Type: image/png\r\n\r\n" . file_get_contents($image_path) . "\r\n--{$boundary}--\r\n";
            wp_remote_post("https://api.telegram.org/bot{$token}/sendPhoto", [
                'headers' => ['Content-Type' => "multipart/form-data; boundary={$boundary}"],
                'body' => $body,
                'timeout' => 30
            ]);
            return;
        }
    }
    // fallback texto
    wp_remote_post("https://api.telegram.org/bot{$token}/sendMessage", ['body' => ['chat_id' => $chat_id, 'text' => $caption, 'parse_mode' => 'HTML']]);
}

function simple_gemini_send_alert_email($post_id, $title, $edit_link) {
    if (!get_option('simple_gemini_alert_enable', 1)) return;
    $to = get_option('simple_gemini_alert_email', get_option('admin_email'));
    if (empty($to)) return;
    $subject = 'Nuevo post automático: ' . $title;
    $message = '<h2>Nuevo post IA</h2><p><strong>' . esc_html($title) . '</strong></p><p><a href="' . esc_url($edit_link) . '">Editar post</a></p>';
    wp_mail($to, $subject, $message, ['Content-Type: text/html; charset=UTF-8']);
}

function simple_gemini_auto_create_post() {
    $sources = trim(
        simple_gemini_get_rss_content() . "\n\n" .
        simple_gemini_get_x_tweets() . "\n\n" .
        simple_gemini_get_ig_content()
    );

    $prompt = $sources 
        ? "Fuentes locales General Roca / Río Negro:\n" . $sources . "\n\nGenera artículo periodístico 500-800 palabras en español. Comienza con 'Título: ...'"
        : "Genera nota sobre actualidad en General Roca / Alto Valle (turismo, economía, clima). Título al inicio.";

    $text = simple_gemini_generate_text($prompt);
    if (is_wp_error($text)) return;

    $lines = explode("\n", trim($text));
    $title = (strpos($lines[0] ?? '', 'Título:') === 0) ? trim(substr($lines[0], 7)) : 'Actualidad IA - ' . date_i18n('d/m/Y');
    $content = implode("\n", array_slice($lines, (strpos($lines[0] ?? '', 'Título:') === 0 ? 1 : 0)));

    $image_id = simple_gemini_generate_image("Imagen destacada realista: " . wp_trim_words($title . ' ' . $content, 15) . " General Roca Río Negro");

    $post_id = wp_insert_post([
        'post_title'     => sanitize_text_field($title),
        'post_content'   => wp_kses_post($content),
        'post_status'    => 'draft',
        'post_author'    => 1,
        'featured_media' => is_wp_error($image_id) ? 0 : $image_id,
    ], true);

    if (is_wp_error($post_id)) return;

    // Categorías automáticas
    $cats = [get_cat_ID('Noticias Automáticas') ?: 1];
    $lower = strtolower($content);
    if (strpos($lower, 'turismo') !== false) $cats[] = get_cat_ID('Turismo') ?: 0;
    if (strpos($lower, 'policial') !== false) $cats[] = get_cat_ID('Policiales') ?: 0;
    if (strpos($lower, 'econom') !== false) $cats[] = get_cat_ID('Economía') ?: 0;
    wp_set_post_categories($post_id, array_filter($cats));

    $edit_link = get_edit_post_link($post_id, '');
    simple_gemini_send_whatsapp_meta($title, $post_id, $edit_link, $image_id);
}

(function(wp) {

    const { useState } = wp.element;
    const { useDispatch, useSelect } = wp.data;

    const GeminiSidebar = () => {
        const [prompt, setPrompt] = useState('');
        const [model, setModel] = useState('gemini-2.5-flash');
        const [loading, setLoading] = useState(false);
        const [error, setError] = useState('');
        const [success, setSuccess] = useState('');

        const { insertBlocks } = useDispatch('core/block-editor');
        const selectedBlockClientId = useSelect(select => 
            select('core/block-editor').getSelectedBlockClientId()
        );

            if (!prompt.trim()) {
                setError('Escribí un prompt primero, loco.');
                return;
            }

            setLoading(true);
            setError('');

                    } else {
                        insertBlocks(newBlock);

                    if (full && res.data.post_id) {
                        setSuccess(`¡Post completo creado! ID: ${res.data.post_id}. Revisalo en borradores.`);
                    } else {
                        setSuccess('Texto generado e insertado.');
                    }
                } else {
                    setError(res.data?.message || 'Algo falló en el servidor.');
                }
            })
            .fail(err => {
                setError(err.message || 'Error de conexión. Revisá tu API key o internet.');
            })
            .always(() => setLoading(false));
        };

        return (
            <PluginSidebar
                name="gemini-sidebar"
                title="Gemini IA - General Roca"
                icon="lightbulb"
            >
                <PanelBody title="Generar con Gemini">
                    <TextareaControl

                            { label: 'Texto + imagen (más lento)', value: 'gemini-2.5-flash-image' }

                        </Button>
